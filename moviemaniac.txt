<script>
    const platforms = [
        { name: "Netflix", url: "https://www.netflix.com", type: "Movies & Series", region: "india" },
        { name: "Prime Video", url: "https://www.primevideo.com", type: "Originals", region: "india" },
        { name: "Disney+ Hotstar", url: "https://www.hotstar.com", type: "Sports & Movies", region: "india" },
        { name: "JioCinema", url: "https://www.jiocinema.com", type: "Free & Premium", region: "india" },
        { name: "SonyLIV", url: "https://www.sonyliv.com", type: "TV Shows", region: "india" },
        { name: "ZEE5", url: "https://www.zee5.com", type: "Regional & Originals", region: "india" },
        { name: "Aha", url: "https://www.aha.video", type: "Telugu / Tamil", region: "india" },
        { name: "Hoichoi", url: "https://www.hoichoi.tv", type: "Bengali", region: "india" },
        { name: "HBO Max", url: "https://www.max.com", type: "Premium", region: "global" },
        { name: "Apple TV+", url: "https://tv.apple.com", type: "High Quality", region: "global" },
        { name: "Crunchyroll", url: "https://www.crunchyroll.com", type: "Anime", region: "global" },
        { name: "Mubi", url: "https://mubi.com", type: "Arthouse", region: "global" }
    ];

    const indiaGrid = document.getElementById('india-grid');
    const globalGrid = document.getElementById('global-grid');
    const searchInput = document.getElementById('searchInput');
    const mainOrb = document.getElementById('mainOrb');

    function renderPlatforms(filter = '') {
        indiaGrid.innerHTML = '';
        globalGrid.innerHTML = '';

        const filtered = platforms.filter(p => 
            p.name.toLowerCase().includes(filter.toLowerCase()) || 
            p.type.toLowerCase().includes(filter.toLowerCase())
        );

        filtered.forEach((p, index) => {
            const card = document.createElement('a');
            card.href = p.url;
            card.target = "_blank";
            card.className = 'card';
            card.innerHTML = `
                <div>
                    <span class="tag">${p.type}</span>
                    <h3>${p.name}</h3>
                </div>
                <div class="arrow">â†’</div>
            `;
            
            if (p.region === 'india') indiaGrid.appendChild(card);
            else globalGrid.appendChild(card);

            setTimeout(() => card.classList.add('visible'), index * 100);
        });
    }

    // Orb Mouse Interaction
    document.addEventListener('mousemove', (e) => {
        const moveX = (window.innerWidth / 2 - e.pageX) / 30;
        const moveY = (window.innerHeight / 2 - e.pageY) / 30;
        mainOrb.style.transform = `translate(${moveX}px, ${moveY}px)`;
    });

    // Click to change color theme
    mainOrb.addEventListener('click', () => {
        const colors = ['#00f2ff', '#7000ff', '#ff007a', '#00ff88'];
        const randomColor = colors[Math.floor(Math.random() * colors.length)];
        document.documentElement.style.setProperty('--accent', randomColor);
        mainOrb.style.boxShadow = `0 0 150px ${randomColor}`;
    });

    searchInput.addEventListener('input', (e) => renderPlatforms(e.target.value));

    // Initial Load
    renderPlatforms();
</script>
